import pandas as pd
import numpy as np
import re

#  
exchange_rates = {
    "$": 92.5,   # 1  = 92.5 .
    "": 100.2,  # 1  = 100.2 .
    "": 1       #    
}

#     (    )
def parse_salary(salary):
    if pd.isna(salary) or " " in str(salary):
        return np.nan
    
    salary = salary.replace(" ", "")  #  
    match = re.match(r"([\$\]?)(\d+)-?(\d+)?", salary)  #    
    if not match:
        return np.nan

    currency, min_val, max_val = match.groups()
    min_val = int(min_val)
    max_val = int(max_val) if max_val else min_val  #   ,   
    avg_salary = (min_val + max_val) / 2  #  

    return avg_salary * exchange_rates.get(currency, 1)  #   

#     (100000  100K)
def format_salary(value):
    if pd.isna(value):
        return np.nan
    if value >= 1000:
        return int(value // 1000)
    return str(value)

#     
def parse_experience(exp):
    if pd.isna(exp):
        return np.nan
    exp = str(exp)
    if " 6 " in exp:
        return 7
    if " 3  6 " in exp:
        return 4.5
    if " 1   3 " in exp:
        return 2
    if " " in exp:
        return 0
    return np.nan

#  
file_path = "dt.csv"
df = pd.read_csv(file_path)

#  
df[""] = df[""].apply(parse_salary)
df[""] = df[""].apply(format_salary)

#   
df[" "] = df[" "].apply(parse_experience)

#  
print(df[["", " "]].head())
